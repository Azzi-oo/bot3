from math import sqrt

MESSAGE = "Добро пожаловать в самую лучшую программу для вычисления квадратного корня из заданного числа"

def calculate_square_root(number):
    """Вычисляет квадратный корень"""
    return sqrt(number)

def calc(your_number):
    """
    Вычисляет квадратный корень из заданного числа и выводит результат на экран

    :param your_number: число, из которого нужно вычислить квадратный корень
    :type your_number: int или float
    """
    if not isinstance(your_number, (int, float)):
        raise TypeError("Number must be a number")
    
    if your_number <= 0:
        raise ValueError("Number must be positive")
     
    square_root = calculate_square_root(your_number)
    print(f"Мы вычислили квадратный корень из введённого вами числа.")
    print(f"Это будет: {square_root}")
    

print(MESSAGE)
calc(25.5)
